import streamlit as st import whois import re import socket from concurrent.futures import ThreadPoolExecutor, as_completed from functools import lru_cache st.set_page_config(page_title="Domain Checker ULTRA FAST", layout="wide") st.title("Domain Checker – ULTRA FAST 🚀") # ============================= # رفع الملف # ============================= uploaded_file = st.file_uploader( "رفع فايل الدومينات (.txt أو .csv)", type=["txt", "csv"] ) # ============================= # إعدادات الفلترة # ============================= keyword = st.text_input("الكلمة (اختياري)") position = st.selectbox("مكان الكلمة", ["any", "start", "end"]) letter_count = st.number_input( "عدد الحروف قبل .com (0 = بلا فلترة)", min_value=0, value=0 ) # ============================= # الامتدادات # ============================= extensions = [ "net","org","io","co","info","biz","app","dev","ai","xyz", "online","site","store","tech","pro","club","live","world", "digital","global","group","media","agency","solutions","cloud" ] # ============================= # فلترة الدومين # ============================= def filter_domain(domain): domain = domain.lower().strip() if not re.fullmatch(r"[a-z]+\.com", domain): return False name_only = domain[:-4] if keyword: kw = keyword.lower() if position == "start" and not name_only.startswith(kw): return False if position == "end" and not name_only.endswith(kw): return False if position == "any" and kw not in name_only: return False if letter_count > 0 and len(name_only) != letter_count: return False return True # ============================= # WHOIS + DNS fallback # ============================= @lru_cache(maxsize=50000) def is_registered(domain): try: w = whois.whois(domain) if w.domain_name: return True except: pass try: socket.gethostbyname(domain) return True except: return False # ============================= # فحص الامتدادات + availability # ============================= def check_domain_exts(domain): name_only = domain[:-4] found_exts = [] # 🔥 تحقق من .com مرة وحدة فقط com_available = not is_registered(domain) with ThreadPoolExecutor(max_workers=8) as ext_executor: futures = { ext_executor.submit(is_registered, f"{name_only}.{ext}"): ext for ext in extensions } for future in as_completed(futures): ext = futures[future] try: if future.result(): found_exts.append(ext) except: pass return domain, found_exts, com_available # ============================= # المعالجة الرئيسية # ============================= if uploaded_file: lines = uploaded_file.readlines() domains = [] for line in lines: decoded = line.decode("utf-8").strip() if decoded: domains.append(decoded.split(",")[0]) filtered = [d for d in domains if filter_domain(d)] st.write(f"عدد الدومينات بعد الفلترة: {len(filtered)}") st.write("---") progress = st.progress(0) results = [] with ThreadPoolExecutor(max_workers=20) as executor: futures = [executor.submit(check_domain_exts, d) for d in filtered] total = len(futures) if futures else 1 for i, future in enumerate(as_completed(futures)): results.append(future.result()) progress.progress((i + 1) / total) # ============================= # عرض النتائج # ============================= for domain, found_exts, available in results: if available: status = "✅ AVAILABLE" else: status = "❌ TAKEN" if found_exts: st.success( f"{domain} → {status} | مسجل فـ: {', '.join(found_exts)}" ) else: st.info( f"{domain} → {status} | غير مسجل فهاد الامتدادات" )
